Theme

ax5dialog themes using six colors defined by the bootstrap is provided.
If you want to change the theme to the dialog you can define the theme attribute at the time of method calls, you can use to define in advance setConfig.

Alert

<script type="text/javascript">
    $(document.body).ready(function () {
        var dialog = new ax5.ui.dialog();
        dialog.setConfig({
            title: "AX5 Dialog theme"
        });

        $('button[data-alert]').click(function () {
            var theme = this.getAttribute("data-alert");
            dialog.alert({
                theme: theme,
                title: 'Alert ' + theme,
                msg: theme + ' color'
            }, function () {
                console.log(this);
            });
        });
    });
</script>
