ax5.util

ax5.util.get_type

Return argument object type 

<script type="text/javascript">
  ax5.util.get_type(1); // "number"
  ax5.util.get_type("1"); // "string"
  ax5.util.get_type([0, 1, 2]); // "array"
  ax5.util.get_type({a: 1}); // "object"
  ax5.util.get_type(function() {}); // "function"
  ax5.util.get_type(document.querySelectorAll("div")); // "nodelist"
  ax5.util.get_type(document.createDocumentFragment()); // "fragment"
</script>

Javascript object type name is not clear. so util.get_type method very useful.

ax5.util.is_[type]

Return argument object type is [type] result. 

<script type="text/javascript">
  // return is window.
  ax5.util.is_window(window);

  // return is element.
  ax5.util.is_element(document.getElementById("#ax5-util-is-type"));

  // return is Object.
  ax5.util.is_object();

  // return is Array.
  ax5.util.is_array();

  // return is Functon.
  ax5.util.is_function();

  // return is String.
  ax5.util.is_string();

  // return is Number.
  ax5.util.is_number();

  // return is nodeList.
  ax5.util.is_nodelist(document.querySelectorAll(".content"));

  // return is undefined.
  ax5.util.is_undefined();

  // return is undefined|''|null.
  ax5.util.is_nothing();
</script>

ax5.util.clone

Return copied new Object 

<script type="text/javascript">
  var a = {a: 1};
  var b = a;
  console.log("(a == b) => " + (a == b));
  b = ax5.util.clone(a);
  console.log("(a == b) => " + (a == b));
</script>

ax5.util.each

You can cycle through the items processed in an Array or an Object using each.
If the object in each function second argument (key, value) is passed, Array in each function second argument (index, Array item) 

<script type="text/javascript">
  var a = {a: "axisj", b: "ax5"};
  var b = [9, 8, 5, 4, 3, 1];

  var po = [];
  po.push("* a");
  ax5.util.each(a, function(k, v) {
    po.push(k + ":" + v + " | " + this);
  });
  po.push("* b");

  ax5.util.each(b, function(index, item) {
    po.push(index + ":" + item + " | " + this);
  });

  console.log(po.join('\n'));
</script>

ax5.util.filter

The first item is delivered to the second argument of the filter function. The second argument is an anonymous function, the result is True, the items will be collected. 

<script type="text/javascript">
  var result, aarray = [5, 4, 3, 2, 1];

  result = ax5.util.filter(aarray, function() {
    return this % 2;
  });
  console.log(result);

  var list = [
    {isdel: 1, name: "ax5-1"}, {name: "ax5-2"}, {isdel: 1, name: "ax5-3"}, {name: "ax5-4"}, {name: "ax5-5"}
  ];
  result = ax5.util.filter(list, function() {
    return (this.isdel != 1);
  });
  console.log(JSON.stringify(result));
</script>
